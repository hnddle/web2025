str.split(" ");
str.includes("Hello");
str.startsWith("Hello");
str.endsWith("World");

주요 메서드

let arr = [1, 2, 3, 4, 5];

arr.push(6);
arr.pop();
arr.unshift(0);
arr.shift();

arr.slice(1, 3);
arr.splice(2, 1, "a", "b");

arr.indexOf(3);
arr.includes(4);

arr.join("-");
arr.reverse();
arr.sort();

arr.forEach(function(item) {
    console.log(item);
});

let doubled = arr.map(x => x * 2);
let filtered = arr.filter(x => x > 2);
let sum = arr.reduce((acc, cur) => acc + cur, 0);

arr.find(x => x > 3);
arr.findIndex(x => x > 3);
arr.every(x => x > 0);
arr.some(x => x > 4);

RegExp 객체

  • 정규표현식을 다루는 객체

기본 사용법

let pattern1 = /hello/i;
let pattern2 = new RegExp("hello", "i");

let text = "Hello World";
pattern1.test(text);
text.match(pattern1);
text.search(pattern1);
text.replace(/hello/i, "Hi");

let email = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
email.test("test@example.com");
