연산자와 제어문

산술 연산자

  • + : 더하기
  • - : 빼기
  • * : 곱하기
  • / : 나누기
  • % : 나머지
  • ** : 거듭제곱
let a = 10;
let b = 3;
console.log(a + b);  
console.log(a - b);  
console.log(a * b);  
console.log(a / b);  
console.log(a % b);  
console.log(a ** b);

대입 연산자

  • = : 대입
  • += : 더한 후 대입
  • -= : 뺀 후 대입
  • *= : 곱한 후 대입
  • /= : 나눈 후 대입
let x = 10;
x += 5;  
x -= 3;  
x *= 2;  
x /= 4;

비교 연산자

  • == : 같다 (타입 변환 후 비교)
  • === : 같다 (타입까지 비교)
  • != : 같지 않다
  • !== : 같지 않다 (타입까지 비교)
  • > : 크다
  • < : 작다
  • >= : 크거나 같다
  • <= : 작거나 같다
console.log(5 == "5");   
console.log(5 === "5");  
console.log(5 != "5");   
console.log(5 !== "5");

논리 연산자

  • && : AND (둘 다 참일 때 참)
  • || : OR (하나라도 참이면 참)
  • ! : NOT (참/거짓 반대)
let age = 25;
let hasLicense = true;

if (age >= 18 && hasLicense) {
    console.log("운전 가능");
}

if (age < 18 || !hasLicense) {
    console.log("운전 불가");
}

조건문 - if문

let score = 85;

if (score >= 90) {
    console.log("A학점");
} else if (score >= 80) {
    console.log("B학점");
} else if (score >= 70) {
    console.log("C학점");
} else {
    console.log("F학점");
}

조건문 - switch문

let day = 3;
let dayName;

switch(day) {
    case 1:
        dayName = "월요일";
        break;
    case 2:
        dayName = "화요일";
        break;
    case 3:
        dayName = "수요일";
        break;
    default:
        dayName = "기타";
}
console.log(dayName);

반복문 - for문

for (let i = 0; i < 5; i++) {
    console.log(i);
}

for (let i = 1; i <= 10; i++) {
    console.log("2 x " + i + " = " + (2 * i));
}

반복문 - while문

let count = 0;
while (count < 5) {
    console.log(count);
    count++;
}

let num = 10;
do {
    console.log(num);
    num--;
} while (num > 0);

break와 continue

for (let i = 0; i < 10; i++) {
    if (i === 5) {
        break;
    }
    console.log(i);
}

for (let i = 0; i < 10; i++) {
    if (i % 2 === 0) {
        continue;
    }
    console.log(i);
}

삼항 연산자

let age = 20;
let message = (age >= 18) ? "성인" : "미성년자";
console.log(message);
